\paragraph{General Notation}
We use caligraphic letters to denote sets, a tilde to denote random variables, and bold lowercase letters to denote
vectors. 
We also adopt the standard convention that $\mathcal{A}^{\mathcal{B}}$ represents the set of all functions from a set $\mathcal{B}$ to a set $\mathcal{A}$ and treat vectors as a function from indexes to real numbers. $\bm{1}$ denotes the vector of ones and $\mathds{1}(bool)$ denotes the indicator function. We use the convention that comparison operators are defined on vector spaces element-wise, that is for $x, y \in \mathbb{R}^n$ then $x \leq y \iff x_i \leq y_i \quad \; \forall \; i \in 1\dots n$. Finally, $\Delta^n$ denotes the standard $n$-simplex while $\Delta_k^n$ denotes the k-capped simplex~\cite{kcapped_ang}.
We assume a probability space with finite outcomes $\Omega = \left\{ 1, \dots , N \right\}$, events $\mathcal{F}$, and
probability measure $\mathbb{P} \in \Delta^N$. We consider a random variable to be any mapping $\tilde{x}\colon \Omega \to \mathbb{R}$. Or equivalently as vectors on $\Omega$. For example, the vector $\bm{x}\in \mathbb{R}^N$ equivalent to $\tilde{x}$ is constructed as
\[
 x_w = \tilde{x}( \omega ), \quad\forall \omega \in \Omega .
\]
We assume the same equivalence for probability distributions.

Quantile of a random variable $\tilde{x} \in \mathbb{X}$ at level $\alpha \in [0,1]$ is any value $\tau\in \mathbb{R}$ such that $\mathbb{P}({\tilde{x} \le \tau}) \ge \alpha$ and $\mathbb{P}({\tilde{x} \ge \tau}) \ge 1-\alpha$. The quantile may not be unique however always lies in the interval $[\quant^-_{\alpha}(\tilde{x}), \quant^+_{\alpha}(\tilde{x})]$ where
\begin{align} \label{eq:quantile-definition}
%\label{eq:upper-quantile-definition}
  \quant_{\alpha}^+(\tilde{x})
  &\;:=\;
    \max\; \left\{\tau \in \bar{\mathbb{R}} \mid \mathbb{P}[\tilde{x} < \tau]  \le \alpha\right\}, 
      %\label{eq:low-quantile-definition}
  &\quant_{\alpha}^-(\tilde{x})
  &\;:=\;
    \min\; \left\{\tau \in \bar{\mathbb{R}} \mid \mathbb{P}[\tilde{x} \le \tau]   \ge \alpha\right\}.
\end{align}
The maximum in~\eqref{eq:quantile-definition} exists because $\mathbb{P}[\tilde{x} < \tau]$ is lower semi-continuous and the minimum in~\eqref{eq:quantile-definition} exists because $\mathbb{P}[\tilde{x} \le \tau]$ is upper semi-continuous with respect to $\tau$. Note that $\quant^+_1(\tilde{x}) = \infty$, $\quant^-_0(\tilde{x}) = -\infty$, otherwise $\quant^+_\alpha, \quant^-_\alpha$ are bounded for all $\alpha \in (0,1)$ and $\tilde{x} \in \mathcal{L}^\infty$.

Monetary risk measures generalize the expectation operator to allow for risk sensitive distinctions in otherwise equivalent distributions~\cite{Follmer2016}.~\emph{Value-at-Risk}~(VaR) is one of the oldest, simplest, and most common risk measures~\cite{Follmer2016,Shapiro2014}. $\varo : \mathbb{X} \to \bar{\mathbb{R}}$ at a risk level $\alpha \in [0,1]$ of a random variable $\tilde{x}\in \mathbb{X}$ is defined as:
\begin{equation}
  \label{eq:var-definition}
\varo_{\alpha}(\tilde{x}) \;:=\;  \quant^+_{ \alpha }(\tilde{x}).  
\end{equation}

\paragraph{$\varphi$-divergence Risk Measures}

$\varphi$-divergences can be defined as 
\[I_\phi(p, q) = \sum_{n=1}^X \phi\left(\frac{p_\omega}{q_\omega}\right)
\] 
for two discrete, absolutely continuous probability distributions $p$ and $q$, and convex $\phi$ such that $\phi(1)=0$. 

Recall that any coherent risk measure can be defined as 
\begin{align}\label{eq:correspondence_coherent_risk}
  \rho(X)=\sup_{Q\in \mathcal{Q}}\mathbb{E}_Q [-X],
\end{align}
where $\mathcal{Q}$ is a subset of all probability measures on $\Omega$~\cite{Follmer2016}(Prop. 2.84). Phi-divergence risk measures can be defined as a coherent risk measure for which $\mathcal{Q}$ contains all $Q$ such that $I_\phi(P, Q) \le \beta$, where $\beta$ is some constant typically dependent on the risk level. Phi-divergence risk measures are equivalent to g-entropic risk measures, which can be seen as the primal representation of these risk measures. All g-entropic risk measures take on the following form for $\beta \in \mathbb{R}$ and any convex \todo{(closed?)} function $g\colon \mathbb{R} \to \mathbb{R}$~\cite{Ahmadi-Javid2011,Ahmadi-Javid2017}: 
\[
\ER_{g,\beta}(\tilde{x}) = \sup_{t > 0, v \in \mathbb{R}} t \left(v + \E {-g^{\star}\left(-\frac{\tilde{x}}{t} + v + \beta \right)}\right),
\] 
where $g^{\star}$ the convex conjugate of the phi-divergence $g$. EVaR, for example, is a well known example of phi-divergence risk measure where the phi-Divergence is the KL-divergence~\cite{Ahmadi-Javid2017}.

\paragraph{Polymatroids}

Let $\mathcal{E}$ be a finite set, and $f : \{\mathcal{X} \mid \mathcal{X} \subseteq \mathcal{E} \} \to \mathbb{R}_+$ a nondecreasing submodular function. That is, for each $\mathcal{A} \subseteq \mathcal{B} \subseteq \mathcal{E}$ we have $f(\mathcal{A}) \leq f(\mathcal{B})$, and for each $\mathcal{A}, \mathcal{B} \subseteq \mathcal{E}$ we have 
\[
f(\mathcal{A}) + f(\mathcal{B}) \geq f(\mathcal{A} \cup \mathcal{B}) + f(\mathcal{A} \cap \mathcal{B}).
\]
We define the polymatroid associated to $f$ to be the following polytope:
\[
  P_f = \left\{ \bm{x} \in \mathbb{R}_+^E  \mid  \sum_{e \in \mathcal{U}} \bm{x}(e) \leq f(\mathcal{U}), \;\; \forall \; \mathcal{U} \subseteq \mathcal{E} \right\}.
\]
This polytope is of special interest for computing risk measures, as we will see later on that all coherent and comonotonic risk measures can be computed by solving
\begin{align}\label{eq:polytope}
  \max_{x}\left\{\bm{x}\tr \bm{p} \mid  \bm{p} \in P_f\right\}
\end{align}
Conventional algorithms such as Algorithm 4, solve \eqref{eq:polytope} through a greedy approach which requires sorting. We will show in \todo{reference section} that for a special class of submodular functions $f$, sorting is not necessary.

\section{Fast Algorithm for VaR}

In this section, we describe a linear time algorithm for VaR.

\begin{algorithm}
  \algrenewcommand\algorithmicrequire{\textbf{Input: vector $x$, pmf $p$ and $\alpha \in [0,1]$}}
  \algrenewcommand\algorithmicensure{\textbf{Output: $\alpha$-quantile of $\tilde{X}$}}
  \caption{Quick Quantile algorithm for VaR}\label{alg:quick_quantile}
  \begin{algorithmic}[1]
    \Require{} \Ensure{}
    \State{$f_0 \gets 1$}
    \State{$b_0 \gets n$ }
    \While{$b_k - f_k \geq 1$}
      \State{$i_k \gets \operatorname{partition}(x, p, f_k, b_k) - 1$  \Comment{Hoare partition scheme}}
    \State{$t_k \gets \sum_{j=1}^n \mathds{1}(x[j] \leq x[i_k]) p[j]$}
      \If{$\alpha < t_k$}
      \State{$b_k \gets i_k$}
      \Else{}
      \State{$f_k \gets i_k + 1$}
    \EndIf{}
    \State $k \gets k + 1$
    \EndWhile{}
    \State{\Return{$x[f_k]$}}
  \end{algorithmic}
\end{algorithm}

\begin{thm}
\cref{alg:quick_cvar} is correct and runs in linear time.

\subsection{General Algorithm}

\subsection{CVaR}
As observed in~\cite{rockefeller1996variational}, the Conditional Value-at-Risk (CVaR) is a convex risk measure that generalizes the Value-at-Risk (VaR) and can be defined as
\begin{equation} \label{eq:primal_cvar}
  \begin{aligned}
    \cvaro_\alpha(\tilde{x})
  &= \min \left\{  \Ex{\bm{\xi}}{ \tilde{x}  } \mid
  \bm{\xi} \leq \frac{1}{1-\alpha} p, \bm{\xi} \in \Delta^n \right\} \\
  &= \min \left\{\bm{\xi}\tr \bm{x} \mid \bm{\xi} \leq \frac{1}{1-\alpha} \bm{p},
     \xi\tr \bm{1} = 1, \bm{\xi} \geq 0\right\}\\
  &= \max \left\{ z - \frac{1}{1-\alpha} \Ex{\bm{p}}{z-\bm{x}}_+ \mid z \in \mathbb{R}\right\} \\
  &= \max \left\{ z - \frac{1}{1-\alpha} \bm{p}\tr(z-\bm{x})_+ \mid z \in \mathbb{R}\right\}.
  \end{aligned}
\end{equation}

\subsection{TVaR}
\begin{algorithm}
  \algrenewcommand\algorithmicrequire{\textbf{Input: vector $x$, pmf $p$ and maximum divergence \beta}}
  \algrenewcommand\algorithmicensure{\textbf{Output: $q$ : new distribution }}
  \caption{Linear Time Algorithm for TVaR}\label{alg:quick_tvar}
  \begin{algorithmic}[1]
    \State{$m \gets \arg\min \left\{ x_i \mid  i = 1, \dots, n \right\}$}
    \State{$\delta \gets \min\left\{\frac{\beta}{2}, 1 - p_m\right\} $}
    \State{$\quant^+ \gets \operatorname{QuickQuantile}(x, q, \delta)$ }
    \State{$q_m \gets q_m + \delta$}
    \State{$\lambda \gets \delta$}
    \For{$i \in 1 \dots n$}
      \If{$q[i] > \quant^+$}
        \State{$\lambda \gets \lambda - q[i]$}
        \State{$q[i] \gets 0$}
      \EndIf{}
    \EndFor{}
    \State{\Return{$\lambda$}}
  \end{algorithmic}
\end{algorithm}

\section{Comonotonicity of Risk}

\begin{definition}
  Two random variables $\tilde{x}$ and $\tilde{y}$ are said to be comonotonic if
  \begin{align*}
    (\tilde{x}(\omega) - \tilde{x}(\omega'))(\tilde{y}(\omega) - \tilde{y}(\omega')) \geq 0 \quad \forall \; \omega, \omega' \in \Omega.
  \end{align*}
\end{definition}
\begin{definition}
  A risk measure is said to be comonotonic if given comonotonic random variables $\tilde{x}$ and $\tilde{y}$
  \begin{align*}
    \rho(\tilde{x} + \tilde{y}) = \rho(\tilde{x}) + \rho(\tilde{y}).
  \end{align*}
\end{definition}

\section{No-sort Polymatroid}

Our first contribution new algorithm for solving the maximum weight Independant Set without sorting for polymatroids. We note that certain additional assumptions have to be satisified on the submodular function as follows.
\begin{assumption}
Assume for $P_f$ that $f(U)$ is such that for all $S_i, S_{ i+1 } \cup E$ where $S_i \cup (i + 1) = S_{ i+1 } $ and $f(S_{ i+1 }) < f(E)$, $f(S_i) + f({i + 1}) = f(S_{ i+1 })$.
\end{assumption}
We note this assumption essentially state that the submodular function must be a piecewise linear function. 
\begin{algorithm}
    \algrenewcommand\algorithmicrequire{\textbf{Input: }Finite $N = [1 \ldots n]$, non-decreasing submodular function $f : 2^N \to \mathbb{R}$ satisfying $f(\emptyset) = 0$ and $f(N) = 1$} 
    \algrenewcommand\algorithmicensure{\textbf{Output: $\max \{f(S) : S\subseteq N\}$}}
  \caption{algorithm}{New algorithm}
 \begin{algorithmic}[1]
     \Require{}
     \Ensure{}
  
    \State{$left \gets 1$}
    \State{$right \gets n$}
    \While{$left  - right \geq 1$}
    \State{$mid \gets partition(left, right, c)$}
    \If{$f(S_{mid}) = f(N)$}
       \State{Let $x_i :=  \begin{cases}0 & \text{for } i \geq mid,\dots \end{cases}$}
    \Else
          \State{Let $x_i :=  \begin{cases}f(i) & \text{for } i \leq mid,\dots \end{cases}$}
    \EndIf{}      
    
    
    
    \EndWhile{}
    \State{$x_{right} \gets f(N) - f(S_{right - 1})$}
\end{algorithmic}
\end{algorithm}
The proof for this algorithm is simple and shows that output is equal to that of Algorithm 4 (the standard greedy algorithm), which has already proved to be both feasible and optimal.
\begin{lem}
Algorithm 1 returns a feasible and optimal solution to the maximum weight independent set for polymatroids.
\end{lem}
\begin{proof}
The optimal solution will always be $f(S_i) - f(S_{i-1})$ assuming the weight $c_i \geq 0$. Our solution is $f(i)$ for $f(S_i) < f(N)$, and by assumption 1 this equals $f(S_i) - f(S_{i - 1})$. If $f(S_i) = f(N)$ and $f(S_{i - 1}) = f(N)$, then $x_i = 0$ by the algorithm.
\end{proof}
