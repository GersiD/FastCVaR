\section{Formal Model}


\paragraph{Value-at-Risk and quantiles}
Quantile of a random variable $\tilde{x} \in \mathbb{X}$ at level $\alpha \in [0,1]$ is any value $\tau\in \mathbb{R}$ such that $\mathbb{P}({\tilde{x}) \le \tau} \ge \alpha$ and $\mathbb{P}({\tilde{x}) \ge \tau} \ge 1-\alpha$. The quantile of a random variable may not be unique and lies in the interval $[\quant^-_{\alpha}(\tilde{x}), \quant^+_{\alpha}(\tilde{x})]$ where
\begin{align} \label{eq:quantile-definition}
%\label{eq:upper-quantile-definition}
  \quant_{\alpha}^+(\tilde{x})
  &\;:=\;
    \max\; \left\{\tau \in \bar{\mathbb{R}} \mid \mathbb{P}[\tilde{x} < \tau]  \le \alpha\right\}, 
      %\label{eq:low-quantile-definition}
  &\quant_{\alpha}^-(\tilde{x})
  &\;:=\;
    \min\; \left\{\tau \in \bar{\mathbb{R}} \mid \mathbb{P}[\tilde{x} \le \tau]   \ge \alpha\right\}.
\end{align}
The maximum in~\eqref{eq:quantile-definition} exists because $\mathbb{P}[\tilde{x} < \tau]$ is lower semi-continuous and the minimum in~\eqref{eq:quantile-definition} exists because $\mathbb{P}[\tilde{x} \le \tau]$ is upper semi-continuous with respect to $\tau$. Note that $\quant^+_1(\tilde{x}) = \infty$, $\quant^-_0(\tilde{x}) = -\infty$ and the quantiles are bounded otherwise for all $\alpha \in [0,1]$.

Monetary risk measures generalize the expectation operator to account for the uncertainty in the distribution.~\emph{Value-at-Risk}~(VaR) is one of the oldest, simplest, and most common risk measures\todo{fixme}~\cite{Follmer2016,Shapiro2014}. Value-at-Risk: $\varo \colon \mathbb{X} \to \bar{\mathbb{R}}$ at a risk level $\alpha \in [0,1]$ of a random variable $\tilde{x}\in \mathbb{X}$ is defined as the largest $1-\alpha$ confidence lower bound on the value of the random variable:
\begin{equation}
  \label{eq:var-definition}
\varo_{\alpha}{\tilde{x}} \;:=\;  \quant^+_{ \alpha }(\tilde{x}).  
\end{equation}
\paragraph{$\varphi$-divergence Risk Measures}


$\varphi$-divergences can be defined as 
\[I_\phi(p, q) = \sum_{n=1}^X \phi\left(\frac{p_\omega}{q_\omega}\right)
\] 
for two discrete probability distributions $p$ and $q$, where $\phi$ is a convex function such that $\phi(1)=0$. 

Recall that any coherent risk measure can be defined as 
\[
\rho(X)=\sup_{Q\in \mathfrak{J}}{\rm E}_{Q},
\]
where $\mathfrak{J}$ is a constrained set of probability measures. Phi-divergence risk measures can be defined as a coherent risk measure for which $\mathfrak{J}$ contains all $Q$ such that $I_\phi(P, Q) \le \beta$, where $\beta$ is some constant typically dependent on the risk level. Phi-divergence risk measures are equivalent to g-entropic risk measures, which can be seen as primal representation of these risk measures. All g-entropic risk measures take on the following form: 
\[
{\ER_{g,\beta}(X) = \displaystyle\mathop{\inf}_{t > 0, v \in \mathfrak{R}} \left\{ t \left[v + {\rm E}_{P}\left(g^{\ast}\left(\frac{X}{t} - v + \beta \right)\right)\right] \right\},
\] 
where $g^*$ the convex function of the phi-divergence $g$ that is in the dual representation. EVaR, for example, is a well known example of phi-divergence risk measure where the phi-Divergence is relative entropy, better known as the KL-divergence. 




\cite{Ahmadi-Javid2017}


\paragraph{Polymatroids}
 

Let $E$ be a finite set, and $f\colon  2^E \to \mathbb{R}_+$ a nondecreasing submodular function. That is, for each $A \subseteq B \subseteq E$ we have $f(A) \leq f(B)$, and for each $A, B \subseteq E$ we have 
\[
f(A) + f(B) \geq f(A \cup B) + f(A \cap B).
\]
We define the polymatroid associated to $f$ to be the following polytope:
\[
P_f = \left\{ \bm{x} \in \mathbb{R}_+^E  \mid  \sum_{e \in U} \bm{x}(e) \leq f(U), \ \forall U \subseteq E \right\}.
\]
This polytope is of special interest for computing risk measures, as we will see later on that all coherent and comonotonic risk measures can be computed by solving the maximum weight independent set problem on a polymatroid. The maximum weight independent set problem on a polymatroid can be defined as
\[
  \max\left\{\bm{c}\tr \bm{x} \mid  \bm{x} \in P_f\right\}
\]
Conventional algorithms such as Algorithm 4, solve this problem through a greedy approach which requires sorting the elements. We will show in the next section that for a special class of submodular functions $f$, we can compute the maximum weight independent set without sorting elements.



\paragraph{General Notation}
We can assume a probability space with outcomes $\Omega = \left\{ 1, \dots , N \right\}$, events $\mathcal{F}$, and
probability function $\mathbb{P} \in \Delta^N$. We consider a random variable to be any mapping $\tilde{x}\colon \Omega \to \mathbb{R}$. Or equivalently as vectors on $\Omega$. For example, the vector $\bm{x}\in \mathbb{R}^N$ equivalent to $\tilde{x}$ is constructed as
\[
 x_w = \tilde{x}( \omega ), \quad\forall \omega \in \Omega .
\]
We assume the same equivalence for probability distributions.


\section{Fast Algorithm for VaR}

In this section, we describe a linear time algorithm for VaR.

\begin{algorithm}
  \algrenewcommand\algorithmicrequire{\textbf{Input: vector $x$, pmf $p$ and $\alpha \in [0,1]$}}
  \algrenewcommand\algorithmicensure{\textbf{Output: $\alpha$-quantile of $\tilde{X}$}}
  \caption{Quick Quantile algorithm for VaR}\label{alg:quick_quantile}
  \begin{algorithmic}[1]
    \Require{} \Ensure{}
    \State{$f_0 \gets 1$}
    \State{$b_0 \gets n$ }
    \While{$b_k - f_k \geq 1$}
      \State{$i_k \gets \operatorname{partition}(x, p, f_k, b_k) - 1$  \Comment{Lomuto partition}}
    \State{$t_k \gets \sum_{j=1}^n \mathds{1}(x[j] \leq x[i_k]) p[j]$}
      \If{$\alpha < t_k$}
      \State{$b_k \gets i_k$}
      \Else{}
      \State{$f_k \gets i_k + 1$}
    \EndIf{}
    \State $k \gets k + 1$
    \EndWhile{}
    \State{\Return{$x[f_k]$}}
  \end{algorithmic}
\end{algorithm}

\begin{thm}
\Cref{alg:quick_cvar} is correct and runs in linear time.

\begin{lem}
Suppose that $\tilde{x}\colon \Omega \to \mathbb{R}$ is a random variable with finite $\Omega$, and $\alpha\in [0,1)$. Then, there exist $\omega_1, \omega_2\in \Omega$ such that $\quant_{\alpha}^+(\tilde{x}) = \tilde{x}(\omega_1)$ and $\quant_{\alpha}^-(\tilde{x}) = \tilde{x}(\omega_2)$ for each $\alpha\in (0,1]$.
\end{lem}
\begin{proof}
Recall the definition:
 \[
  \quant_{\alpha}^+(\tilde{x})
  \;:=\;
    \max\; \left\{\tau \in \bar{\Real} \mid \P{ \tilde{x} < \tau } \le \alpha\right\}, 
 \] 
Assume the random variable $ x $ has $ n $ unique values, such that $ x_1 < x_2 < \cdots < x_n $.  We can assume this without loss of generality because VaR is law invariant~\cite{Follmer2016}.  

Suppose for the sake of contradiction that there is no $\omega\in \Omega $ such that $\quant^+_{\alpha}(\tilde{x}) = \tilde{x}(\omega)$. That may mean:
\begin{enumerate}
\item[Case 1] $\exists x_{i}, x_{i + 1} \in \tilde{x}(\Omega)$ such that $\quant^+_{\alpha}(\tilde{x}) \in (x_{i}, x_{i + 1})$.
Let $\tau\opt $ is optimal in \eqref{eq:var-definition} (max definition) and therefore $\tau\opt  = \quant^+_{\alpha}(\tilde{x})  $. Then, let $\tau' = \tau\opt  + \frac{1}{2}(x_{i+1} - \tau\opt)$. Because $\tau' > \tau\opt $ and $\tau'$ is feasible in \eqref{eq:var-definition}, $\P{ \tilde{x} < \tau' } =  \P{ \tilde{x} < \tau\opt  }\le \alpha$, $\tau\opt $ cannot be optimal.
\item[Case 2] $\quant^+_{\alpha}(\tilde{x}) < \min \tilde{x}$ 
  
  Let $\tau\opt $ is optimal in \eqref{eq:var-definition} (max definition) and therefore $\tau\opt  = \quant^+_{\alpha}(\tilde{x})  $. Then, let $\tau' = \tau\opt + \frac{1}{2}(x_{1} - \tau\opt)$. Because $\tau' > \tau\opt $ and $\tau'$ is feasible in \eqref{eq:var-definition}, $\P{ \tilde{x} < \tau' } =  \P{ \tilde{x} < \tau\opt  }\le \alpha$, $\tau\opt $ cannot be optimal.

\item[Case 3]  $\quant^+_{\alpha}(\tilde{x}) > \max \tilde{x}$ 

  

  Let $\tau\opt $ is optimal in \eqref{eq:var-definition} (max definition) and therefore $\tau\opt  = \quant^+_{\alpha}(\tilde{x})$. Recall $\alpha \in [0,1)$, $\P{\tilde{x} < \tau\opt} = 1 \leq \alpha < 1$.
  
  % New Proof 2: Recall that $\alpha$ is less than 1 and  $\quant^+_{\alpha}(\tilde{x})$ is bounded for all admissible $\alpha$ values. Let $\tau\opt $ is optimal in \eqref{eq:var-definition} (max definition) and therefore $\tau\opt  = \quant^+_{\alpha}(\tilde{x})  $. Then, let $\tau' = \tau\opt + 1$. Because $\tau' > \tau\opt $ and $\tau'$ is feasible in \eqref{eq:var-definition}, $\P{ \tilde{x} < \tau' } =  \P{ \tilde{x} < \tau\opt  }\le \alpha$, $\tau\opt $ cannot be optimal.

  % Old Proof: Then $ P[x < \quant_{\alpha}^+(\tilde{x})] $ is greater than or equal to $ P[x < x_{n}] $. Remember that $\alpha$ must be less than 1 and $\quant_{\alpha}^+(\tilde{x})$ is bounded for all $\alpha$ less than 1 and greater than or equal to 0, therefore $ P[x < \quant_{\alpha}^+(\tilde{x})] $ is less than 1. Then, there must exist $ y $ in the range $ [\quant_{\alpha}^+(\tilde{x}), \infty) $ such that $ P[x = y] > 0 $ . This is impossible by definition. 
\end{enumerate}
\end{proof}

 Loop Invariant: Before and after every iteration $\quant^+$ will remain in the array $x$ at an index $i$ such that $ front \leq i \leq back $.

Then $P[x < x[\tau\opt]] \geq tail > \alpha$, which means that $x[\tau\opt]$ is not feasible with \eqref{eq:var-definition}. %However, if $\tau\opt $ such that $\tau\opt \leq ind$ , then $P[x < x[\tau\opt]]  < tail > \alpha$, meaning that $x[\tau\opt]$ is potentially feasible by \eqref{eq:var-definition}, so by Lemma 1 $\quant^+$ must remain in this range.% 
 Since from Lemma 1 we know that $\quant^+$ must be in $x$, the loop invariant must hold.

\item[Case 2]: Tail is selected such that $\alpha > tail$. In this case $front$ is set to $ind + 1$. Assume there exists $\tau\opt $ such that $\tau\opt \leq $x[ind]$ $, and $x[\tau\opt]=\quant^+$. If $x[\tau\opt] =  x[\tau']$, then from the partitioning scheme we this is the optimal $\quant^+$ since 
 
 \item[Case 3]: $front - back = 0$.
 Two scenarios here. If $x$ contains more than one element, all the elements are no longer in the range $[front, back]$ could not have been $\quant_{\alpha}^+(\tilde{x})$ (see Case 1 and Case 2), and $\quant_{\alpha}^+(\tilde{x})$ must exist in x per lemma 1. Therefore this element that will be returned is $\quant_{\alpha}^+(\tilde{x})$.  If $x$ had one element to begin with, $\quant_{\alpha}^+(\tilde{x})$ equals this element for all admissible alpha. 
 \end{proof}
 \end{lem}
\begin{thm}
At every iteration, the algorithm will always hit case 1 or case 2 assuming $back - front >= 1$. If $alpha < tail$, $back$ will always get smaller assuming it is in the range $(front, back$. If $alpha \geq tail$ , then  $back - front$ will not get smaller if $ind = front$. However, in this case the elements at $front$ and $back$ will have been swapped in the partition scheme, such that at next iteration $ind$ will not be equal to $front$ (as the pivot selected will now no longer be the smallest element in the list). 
\end{thm}

  CVAR can be computed by solving the following optimization problem $\min_{Q \in \mathcal{Q}} \mathbb{E}_Q[X]$, where $\mathcal{Q} = \left\{Q \in \Delta^\Omega \mid Q(\omega) < \frac{1}{1-\alpha} P(\omega) \quad \forall \omega \in \Omega\right\}$.

\begin{lem}
  For $\omega \in \Omega$ we can construct a feasible $Q \in \mathcal{Q}$ such that $Q(\omega) = \min\left\{\frac{P(\omega)}{1-a}, 1\right\}$.
\end{lem}
\begin{proof}
  Two possible scenarios exist. 
  Case 1: $P(A) \geq 1 - \alpha$. In this case, Q(A) must equal 1 since $(1/(1 - \alpha))* P(A) > 1 $. Thus every element in $A$ should be scaled by $1/P(A)$ if in $A$ and set to 0 otherwise. Since  $P(A) \geq 1 - \alpha$, $1/P(A) \leq 1/(1-\alpha)$  
  Case 2: $P(A) < (1 - \alpha)$. In this scenario, $Q(A) \leq (1/(1 - \alpha)) \cdot  P(A) < 1$. For any given A, We can construct a Q(A) such that $Q(A) = (1/(1 - \alpha)) \cdot  P(A)$. This can be done by setting $Q(a)$ to $(1/(1 - \alpha)) \cdot P(a)$ for all a in A and $P(a) \cdot  d$ otherwise such that $(1/(1 - \alpha)) * P(A) + d*P(B) = 1$, where B contains all elements not in $A$. It is provable that $d \leq (1/(1 - \alpha))$, which is required for the constructed \(Q\) to be feasible. For all feasible $\alpha \geq 0$, $(1/(1 - \alpha)) \ge 1$. Since $P(A) + P(B) = 1$,  $d \le 1$ in order for $(1/(1 - \alpha)) \cdot  P(A) + d\cdot P(B) = 1$ to hold. Since $1 \leq 1/(1-\alpha)$, $d\le \leq (1/(1 - \alpha))$.
\end{proof}

\subsection{CVaR}
As observed in~\cite{rockefeller1996variational}, the Conditional Value-at-Risk (CVaR) is a convex risk measure that generalizes the Value-at-Risk (VaR) and can be defined as
\begin{equation} \label{eq:primal_cvar}
  \begin{aligned}
    \cvaro_\alpha(\tilde{x})
  &= \min \left\{  \Ex{\bm{\xi}}{ \tilde{x}  } \mid
  \bm{\xi} \leq \frac{1}{1-\alpha} p, \bm{\xi} \in \Delta^n \right\} \\
  &= \min \left\{\bm{\xi}\tr \bm{x} \mid \bm{\xi} \leq \frac{1}{1-\alpha} \bm{p},
     \xi\tr \bm{1} = 1, \bm{\xi} \geq 0\right\}\\
  &= \max \left\{ z - \frac{1}{1-\alpha} \Ex{\bm{p}}{z-x}_+ \mid z \in \mathbb{R}\right\}.
  \end{aligned}
\end{equation}

\section{No-sort Polymatroid}

Our first contribution new algorithm for solving the maximum weight Independant Set without sorting for polymatroids. We note that certain additional assumptions have to be satisified on the submodular function as follows.
\begin{assumption}
Assume for $P_f$ that $f(U)$ is such that for all $S_i, S_{ i+1 } \cup E$ where $S_i \cup (i + 1) = S_{ i+1 } $ and $f(S_{ i+1 }) < f(E)$, $f(S_i) + f({i + 1}) = f(S_{ i+1 })$.
\end{assumption}
We note this assumption essentially state that the submodular function must be a piecewise linear function. 
\begin{algorithm}
    \algrenewcommand\algorithmicrequire{\textbf{Input: }Finite $N = [1 \ldots n]$, non-decreasing submodular function $f : 2^N \to \mathbb{R}$ satisfying $f(\emptyset) = 0$ and $f(N) = 1$} 
    \algrenewcommand\algorithmicensure{\textbf{Output: $\max \{f(S) : S\subseteq N\}$}}
  \caption{algorithm}{New algorithm}
 \begin{algorithmic}[1]
     \Require{}
     \Ensure{}
  
    \State{$left \gets 1$}
    \State{$right \gets n$}
    \While{$left  - right \geq 1$}
    \State{$mid \gets partition(left, right, c)$}
    \If{$f(S_mid) = f(N)$}
       \State{Let $x_i :=  \begin{cases}0 & \text{for } i \geq mid,\dots \end{cases}$}
    \Else
          \State{Let $x_i :=  \begin{cases}f(i) & \text{for } i \leq mid,\dots \end{cases}$}
    \EndIf{}      
    
    
    
    \EndWhile{}
    \State{$x_right \gets f(N) - f(S_{right - 1})$}
\end{algorithmic}
\end{algorithm}
The proof for this algorithm is simple and shows that output is equal to that of Algorithm 4 (the standard greedy algorithm), which has already proved to be both feasible and optimal.
\begin{lem}
Algorithm 1 returns a feasible and optimal solution to the maximum weight independent set for polymatroids.
\end{lem}
\begin{proof}
The optimal solution will always be $f(S_i) - f(S_{i-1})$ assuming the weight $c_i \geq 0$. Our solution is $f(i)$ for $f(S_i) < f(N)$, and by assumption 1 this equals $f(S_i) - f(S_{i - 1})$. If $f(S_i) = f(N)$ and $f(S_{i - 1}) = f(N)$, then $x_i = 0$ by the algorithm.
\end{proof}
