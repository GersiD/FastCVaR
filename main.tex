\section{Preliminaries}

Before describing CVaR and its solution we define the basic notation we use in the paper. 
We use calligraphic letters to denote sets and a tilde to denote random variables. 
We also adopt the standard convention that $\mathcal{A}^{\mathcal{B}}$ represents the set of all functions from a set $\mathcal{B}$ to a set $\mathcal{A}$ and 
treat vectors as a function from indexes to real numbers. $\mathbf{1}$ denotes the vector of ones and $\mathds{1}(bool)$ denotes the indicator function. We use the convention that comparison operators are defined on vector spaces element-wise, that is for $x, y \in \Real^n$ then $x \leq y \iff x_i \leq y_i \quad \; \forall \; i \in 1\dots n$.
Finally, $\Delta^n$ denotes the standard $n$-simplex while $\Delta_k^n$ denotes the k-capped simplex~\cite{kcapped_ang}.

\paragraph{Value-at-Risk and Quantiles}

Quantile of a random variable $\tilde{x} \in \mathbb{X}$ at level $\alpha \in [0,1]$ is any value $\tau\in \Real$ such that $\P{\tilde{x} \le \tau} \ge \alpha$ and $\P{\tilde{x} \ge \tau} \ge 1-\alpha$. The quantile of a random variable may not be unique and lies in the interval $[\quant^-_{\alpha}(\tilde{x}), \quant^+_{\alpha}(\tilde{x})]$ where
%\begin{subequations} 
\begin{align} \label{eq:quantile-definition}
%\label{eq:upper-quantile-definition}
  \quant_{\alpha}^+(\tilde{x})
  &\;:=\;
    \max\; \left\{\tau \in \bar{\Real} \mid \P{ \tilde{x} < \tau } \le \alpha\right\}, 
      %\label{eq:low-quantile-definition}
  &\quant_{\alpha}^-(\tilde{x})
  &\;:=\;
    \min\; \left\{\tau \in \bar{\Real} \mid \P{ \tilde{x}  \le \tau } \ge \alpha\right\}.
\end{align}
%\end{subequations}
The maximum in \eqref{eq:quantile-definition} exists because $\tau \mapsto \P{\tilde{x} < \tau}$ is lower semi-continuous and the minimum in \eqref{eq:quantile-definition} exists because $\tau \mapsto \P{\tilde{x} \le \tau}$ is upper semi-continuous. Note that $\quant^+_1(\tilde{x}) = \infty$, $\quant^-_0(\tilde{x}) = -\infty$ and the quantiles are bounded otherwise for all $\alpha \in [0,1]$.

Monetary risk measures generalize the expectation operator to account for the uncertainty in the distribution. \emph{Value-at-Risk}~(VaR) is one of the oldest, simplest, and most common risk measures~\cite{Follmer2016, Shapiro2014}. Value-at-Risk $\varo \colon \mathbb{X} \to \bar{\Real}$ at a risk level $\alpha \in [0,1]$ of a random variable $\tilde{x}\in \mathbb{X}$ is defined as the largest $1-\alpha$ confidence lower bound on the value of the random variable:
\begin{equation}
  \label{eq:var-definition}
\var{\alpha}{\tilde{x}} \;:=\;  \quant^+_{ \alpha }(\tilde{x}).  
\end{equation}

\begin{lem}
Suppose that $\tilde{x}\colon \Omega \to \Real$ is a random variable and $\Omega$ is finite. Then, there exist $\omega_1, \omega_2\in \Omega$ such that $\quant_{\alpha}^+(\tilde{x}) = \tilde{x}(\omega_1)$ for each $\alpha\in [0,1)$ and $\quant_{\alpha}^-(\tilde{x}) = \tilde{x}(\omega_2)$ for each $\alpha\in (0,1]$.
\end{lem}
\begin{proof}
Let
 \[
   f(\tau) = \P{\tilde{x} < \tau}
 \]


 Let $x \in \Real^n$, such that $x_1 < x_2 < \cdots < x_n$. Then for any two elements $x_i$ and $x_{i+1}$, any value $y$ in the range $f(x_i, x_{i+1})$ will have a probability of zero. This means that $f(x_{i + 1}) equals f(y)$ for all $y$ in the specified range, as the probability of $x$ being less than any $y$ and $x_{i+1}$ means it only takes on the values of elements ranging from $x_1$ to $x_i$, which has a fixed probability of occurring. For $y$ in the range $(-\infty, x_1]$, the value of $f(y)$ must be 0 as there is no probability of any element occurring that is smaller than $x_1$, so the random variable  $x$ must always be $\geq x_1$. Likewise, for $y$ in the range $[x_n, \infty)$, the value of $f(y)$ must be 1 as there is no probability of any element occurring that is larger than $x_n$. 


There is thus $n + 1$ values that $f(\tau)$ can take on, including 0, 1, and $n - 1$ values between 0 and 1. Let us define these values as a set $S$. 

Now let's refer again to Definition 1. The term  $\P{\tilde{x} < \tau}$ , which is equal to $f(\tau)$ must take on the maximum value in the set $S$ which is less than or equal to $\alpha$. $\P{\tilde{x} < \tau}$  can never take the value 1, however, because 1 is not an admissible value for $\alpha$. For every value $v$ in set $S$ other than 1, the domain for which $v$ holds on the function $f$ is either $(-\infty, x_1]$ if $v 
 equals  0$, and $(x_i, x_{i+1})$ otherwise. The max element in the domain for every $v$ is an outcome of random variable $x$, therefore for Definition 1 to be satisfied, $\tau$  would have to be one of these values.
\end{proof}

\subsection{Quick Quantile}

\begin{algorithm}
  \floatname{algorithm}{Quick Quantile}
  \algrenewcommand\algorithmicrequire{\textbf{Input: Random variable $\tilde{X}$ with domain $x$ and probability mass function $p$ and $\alpha \in [0,1]$}}
  \algrenewcommand\algorithmicensure{\textbf{Output: $\alpha$-quantile of $\tilde{X}$}}
  \caption{}\label{alg:quick_quantile}
  \begin{algorithmic}[1]
    \Require 
    \Ensure 
    \If{$n = 1$}
    \State \Return $x$
    \EndIf
    \State $pivot \gets rand(1:n)$
    \State $tail \gets \sum_{i=0}^n \mathds{1}(x[i] \leq x[pivot]) p[i]$
    \If{$\alpha \leq tail$}
    \State \Return $qq(vals[\mathds{1}(x[i] \leq x[pivot])], p[\mathds{1}(x[i] \leq x[pivot])], \alpha)$
    \Else
    \State \Return $qq(vals[\mathds{1}(x[i] > x[pivot])], p[\mathds{1}(x[i] > x[pivot])], \alpha - tail)$
    \EndIf
  \end{algorithmic}
\end{algorithm}

\begin{algorithm}
  \floatname{algorithm}{Quick Quantile Loopy}
  \algrenewcommand\algorithmicrequire{\textbf{Input: Random variable $\tilde{X}$ with domain $x$ and probability mass function $p$ and $\alpha \in [0,1]$}}
  \algrenewcommand\algorithmicensure{\textbf{Output: $\alpha$-quantile of $\tilde{X}$}}
  \caption{}\label{alg:quick_quantile_loop}
  \begin{algorithmic}[1]
    \Require 
    \Ensure 
    \While{$n > 1$}
      \State $pivot \gets rand(1:n)$
      \State $tail \gets \sum_{i=0}^n \mathds{1}(x[i] \leq x[pivot]) p[i]$
      \If{$\alpha \leq tail$}
      \State $n \gets \sum_{i=0}^n \mathds{1}(x[i] \leq x[pivot])$
      \Else
      \State $n \gets \sum_{i=0}^n \mathds{1}(x[i] > x[pivot])$
      \State $\alpha \gets \alpha - tail$
      \EndIf
    \EndWhile
  \end{algorithmic}
\end{algorithm}

\section{k-capped simplex}

\begin{equation}
  \label{eq:k_capped_simplex}
  \psi(x, k) = 
  \begin{mprog}
    \minimize{\xi \in \Real^n} \lvert \lvert \xi - x \rvert \rvert_2^2
    \stc \xi \in \Delta_k^n
  \end{mprog}
  =
  \begin{mprog}
    \minimize{\xi \in \Real^n} \lvert \lvert \xi - x \rvert \rvert_2^2
    \stc \xi\tr \mathbf{1} \leq k
    \cs \xi \geq 0   
  \end{mprog}
\end{equation}
